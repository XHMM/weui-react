import { Playground, Props } from 'docz'
import { TopTipsProvider, useTopTips, WeUIButton } from '../../build'
import "../../build/index.css"


## Props
|params|type|default|
|:---:|:---:|:---:|
|content|string|""|
|config| { type: 'warn' &#124; 'primary', duration: number } | { type: 'warn', duration: 2000 }|

## Example
<Playground>
{
  () => {
    const App = function({children}) {
      return <TopTipsProvider><Cmp /></TopTipsProvider>
    }
    
    const Cmp = function() {
        const showTopTips = useTopTips();
        return (<>
            <WeUIButton onClick={() => {showTopTips("content", { type: 'primary', duration: 5000})}} >show primary</WeUIButton>        
            <WeUIButton type="warn" onClick={() => {showTopTips("content")}}>show warning</WeUIButton>
        </>)
    }
    
    return <App />
  }
}

</Playground>
