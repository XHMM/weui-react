import { Playground, Props } from 'docz'
import { WeUIForm, WeUIFormRadioGroup } from '../../build'
import "../../build/index.css"


### `Notes`
difference between this component and official radio weui styles is : as example below, this component show title on left but official not has this. 

### `Props`
<Props of={WeUIFormRadioGroup} />

## Usage
<Playground>
    {
      () => {
        const [sex, setSex] = React.useState("male");
        
        function onSexChange(value) {
          setSex(value);
        }
        
        return (
          <WeUIForm>
            <WeUIFormRadioGroup 
                title="sex change" 
                onChange={onSexChange} 
                value={sex} 
                data={ [{label: 'male', value:'male'}, {label: 'female', value: 'female'}] } 
            />
          </WeUIForm>
        )
      }  
    }
</Playground>
