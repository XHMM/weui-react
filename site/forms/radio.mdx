<Playground>
    {
      () => {
        const [sex, setSex] = React.useState("male");
        
        function onSexChange(ev) {
          setSex(ev.target.value);
        }
        
        return (
          <WeUIForm>
            <WeUIFormRadioGroup title="sex change" onChange={onSexChange} value={sex}>
              <WeUIFormRadio label="male" value="male" />
              <WeUIFormRadio label="female" value="female" />
            </WeUIFormRadioGroup>
          </WeUIForm>
        )
      }  
    }
</Playground>
